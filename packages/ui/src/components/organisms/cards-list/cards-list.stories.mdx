export const PlaygroundTemplate = ({ cardsNumber, maxCardWidth, ...args }) => {
  const cards = Array.from({ length: cardsNumber });
  return (
    <CardsList maxCardWidth={maxCardWidth}>
      {cards.map((card) => (
        <CardStories.playground {...args} key={card} />
      ))}
    </CardsList>
  );
};

# Cards List
